export const ERC20GasPaymentPlayground = () =>
{
    const [state, setState] = useState("initial")
    const [isLoading, setIsLoading] = useState(false)
    const [address, setAddress] = useState("");
    const [transactionHash, setTransactionHash] = useState("");
    const [userOpId, setUserOpId] = useState("");
    const [balance, setBalance] = useState("")
    const [privateKey, setPrivateKey] = useState(null)
    const [status, setStatus] = useState("");
    const [copyStatus, setCopyStatus] = useState("")
    const [refreshing, setRefreshing] = useState(false)
    const baseAPIUrl = "http://localhost:3001"

    const handleCreateAccount = async () =>
    {
        setIsLoading(true)
        setStatus("")
        try {
            const response = await fetch(`${baseAPIUrl}/api/create-account`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
            })
            const data = await response.json()
            if (data.error) throw new Error(data.error)
            setPrivateKey(data.privateKey)
            setAddress(data.address)
            const bal = await fetchBalance(data.privateKey)
            setBalance(bal)
            setState("account-created")
        } catch (e) {
            setStatus("Error creating account")
        }
        setIsLoading(false)
    }

    const fetchBalance = async (privKey) =>
    {
        try {
            const response = await fetch(`${baseAPIUrl}/api/fetch-balance-erc20`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ privateKey: privKey })
            })
            const data = await response.json()
            if (data.error) throw new Error(data.error)
            return data.balance
        } catch (e) {
            return "0.00"
        }
    }

    const handleSendTransaction = async () =>
    {
        setIsLoading(true)
        setStatus("")
        try {
            const response = await fetch(`${baseAPIUrl}/api/send-transaction-erc20`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ privateKey })
            })
            const data = await response.json()
            if (data.error) throw new Error(data.error)
            setUserOpId(data.gelatoId)
            setTransactionHash(data.txHash)
            const bal = await fetchBalance(privateKey)
            setBalance(bal)
            setState("transaction-sent")
            setStatus(`Transaction sent! Gelato ID: ${data.gelatoId}`)
        } catch (e) {
            setStatus(`Error: ${e.message}`)
        }
        setIsLoading(false)
    }

    const truncateAddress = (addr) =>
    {
        return `${addr.slice(0, 6)}...${addr.slice(-4)}`
    }

    const WalletIcon = () => (

        <svg
            width="24"
            height="24"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <path d="M19 7V4a1 1 0 0 0-1-1H5a2 2 0 0 0 0 4h15a1 1 0 0 1 1 1v4h-3a2 2 0 0 0 0 4h3a1 1 0 0 0 1-1v-2a1 1 0 0 0-1-1" />
            <path d="M3 5v14a2 2 0 0 0 2 2h15a1 1 0 0 0 1-1v-4" />
        </svg>
    )

    const SendIcon = () => (

        <svg
            width="20"
            height="20"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <path d="m22 2-7 20-4-9-9-4Z" />
            <path d="M22 2 11 13" />
        </svg>
    )

    const ExternalLinkIcon = () => (

        <svg
            width="16"
            height="16"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <path d="M15 3h6v6" />
            <path d="M10 14 21 3" />
            <path d="M18 13v6a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2V8a2 2 0 0 1 2-2h6" />
        </svg>
    )

    const LoaderIcon = () => (

        <svg
            width="20"
            height="20"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
            className="animate-spin"
        >
            <path d="M21 12a9 9 0 1 1-6.219-8.56" />
        </svg>
    )

    const ChevronRightIcon = () => (

        <svg
            width="20"
            height="20"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <path d="m9 18 6-6-6-6" />
        </svg>
    )

    const CopyIcon = () => (

        <svg
            width="16"
            height="16"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <rect x="9" y="9" width="13" height="13" rx="2" />
            <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1" />
        </svg>
    )

    const RefreshIcon = () => (

        <svg
            width="16"
            height="16"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
        >
            <path d="M23 4v6h-6" />
            <path d="M1 20v-6h6" />
            <path d="M3.51 9a9 9 0 0 1 14.13-3.36L23 10M1 14l4.36 4.36A9 9 0 0 0 20.49 15" />
        </svg>
    )

    const buttonStyles = {
        primary: {
            background: 'linear-gradient(to right, #FF3B57, #ec4899)',
            color: 'white',
            border: 'none',
            padding: '16px 32px',
            borderRadius: '12px',
            fontSize: '18px',
            fontWeight: '500',
            cursor: 'pointer',
            transition: 'all 0.3s ease',
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            gap: '8px',
            boxShadow: '0 10px 25px rgba(255, 59, 87, 0.2)',
        },
        secondary: {
            background: 'rgba(255, 59, 87, 0.1)',
            color: '#FF3B57',
            border: '1px solid rgba(255, 59, 87, 0.3)',
            padding: '12px 24px',
            borderRadius: '8px',
            fontSize: '16px',
            cursor: 'pointer',
            transition: 'all 0.3s ease',
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            gap: '8px',
        },
        ghost: {
            background: 'transparent',
            color: '#9ca3af',
            border: 'none',
            padding: '8px 16px',
            borderRadius: '6px',
            cursor: 'pointer',
            transition: 'all 0.3s ease',
        }
    }

    const cardStyles = {
        background: 'rgba(0, 0, 0, 0.7)',
        border: '1px solid rgba(255, 59, 87, 0.2)',
        borderRadius: '16px',
        backdropFilter: 'blur(12px)',
        boxShadow: '0 25px 50px rgba(255, 59, 87, 0.05)',
    }

    const handleCopy = async (text) =>
    {
        try {
            await navigator.clipboard.writeText(text)
            setCopyStatus("Copied!")
            setTimeout(() => setCopyStatus(""), 1200)
        } catch { }
    }

    const handleRefreshBalance = async () =>
    {
        if (!privateKey) return
        setRefreshing(true)
        const bal = await fetchBalance(privateKey)
        setBalance(bal)
        setRefreshing(false)
    }

    return (

        <div style={{
            minHeight: '100vh',
            background: 'black',
            position: 'relative',
            overflow: 'hidden',
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
        }}>
            <div style={{
                position: 'absolute',
                inset: 0,
                background: 'linear-gradient(135deg, black 0%, rgba(255, 59, 87, 0.1) 50%, black 100%)',
            }}></div>

            <div style={{
                position: 'absolute',
                top: '25%',
                left: '40px',
                width: '128px',
                height: '128px',
                borderRadius: '50%',
                background: 'rgba(255, 59, 87, 0.1)',
                filter: 'blur(48px)',
            }}></div>
            <div style={{
                position: 'absolute',
                bottom: '25%',
                right: '40px',
                width: '160px',
                height: '160px',
                borderRadius: '50%',
                background: 'rgba(255, 59, 87, 0.1)',
                filter: 'blur(48px)',
            }}></div>

            {state !== "initial" && (
                <>
                    <div style={{
                        position: 'absolute',
                        top: '24px',
                        left: '24px',
                        zIndex: 10,
                    }}>
                        <div style={{
                            display: 'flex',
                            alignItems: 'center',
                            gap: '10px',
                            background: 'rgba(0, 0, 0, 0.4)',
                            backdropFilter: 'blur(12px)',
                            padding: '10px 16px',
                            borderRadius: '8px',
                            border: '1px solid rgba(255, 59, 87, 0.2)',
                        }}>
                            <div style={{ color: '#FF3B57' }}>
                                <WalletIcon />
                            </div>
                            <span style={{ fontWeight: '600', color: 'white', letterSpacing: '-0.025em' }}>
                                Smart Wallet SDK
                            </span>
                        </div>
                    </div>

                    <div style={{
                        position: 'absolute',
                        top: '24px',
                        right: '24px',
                        zIndex: 10,
                    }}>
                        <div style={{
                            ...cardStyles,
                            padding: '12px',
                            minWidth: '200px',
                        }}>
                            <div style={{ display: 'flex', flexDirection: 'column', gap: '6px' }}>
                                <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
                                    <span style={{
                                        border: '1px solid rgba(255, 59, 87, 0.3)',
                                        color: '#FF3B57',
                                        background: 'rgba(255, 59, 87, 0.1)',
                                        padding: '4px 10px',
                                        borderRadius: '4px',
                                        fontFamily: 'monospace',
                                        fontSize: '14px',
                                        display: 'inline-block',
                                        textAlign: 'center',
                                        width: '100%',
                                    }}>
                                        {truncateAddress(address)}
                                    </span>
                                    <button
                                        style={{
                                            background: 'transparent',
                                            border: 'none',
                                            color: '#9ca3af',
                                            cursor: 'pointer',
                                            padding: '4px',
                                            display: 'flex',
                                            alignItems: 'center',
                                        }}
                                        onClick={() => handleCopy(address)}
                                    >
                                        <CopyIcon />
                                    </button>
                                    {copyStatus && (
                                        <span style={{ color: '#10b981', fontSize: '12px', marginLeft: '4px' }}>{copyStatus}</span>
                                    )}
                                </div>
                                <div style={{
                                    fontSize: '14px',
                                    color: '#9ca3af',
                                    display: 'flex',
                                    alignItems: 'center',
                                    justifyContent: 'space-between',
                                    gap: '6px',
                                }}>
                                    <span>Balance:</span>
                                    <span style={{ color: 'white', fontFamily: 'monospace', fontWeight: '500', display: 'flex', alignItems: 'center', gap: '6px' }}>
                                        {balance} USDC
                                        <button
                                            style={{
                                                background: 'transparent',
                                                border: 'none',
                                                color: refreshing ? '#FF3B57' : '#9ca3af',
                                                cursor: refreshing ? 'not-allowed' : 'pointer',
                                                padding: '2px',
                                                marginLeft: '4px',
                                                display: 'flex',
                                                alignItems: 'center',
                                                transition: 'color 0.2s',
                                            }}
                                            onClick={handleRefreshBalance}
                                            disabled={refreshing}
                                        >
                                            <span className={refreshing ? 'animate-spin' : ''}><RefreshIcon /></span>
                                        </button>
                                    </span>
                                </div>
                            </div>
                        </div>
                    </div>
                </>
            )}

            <div style={{ width: '100%', maxWidth: '672px', padding: '0 6px', zIndex: 10, margin: '0 auto' }}>
                <div style={{ ...cardStyles, padding: '64px 40px', display: 'flex', flexDirection: 'column', alignItems: 'center', marginBottom: '4px' }}>
                    {state === "initial" && (
                        <div style={{ textAlign: 'center', width: '100%' }}>
                            <div style={{ marginBottom: '24px' }}>
                                <h1 style={{
                                    fontSize: '48px',
                                    fontWeight: 'bold',
                                    background: 'linear-gradient(to right, white, #d1d5db)',
                                    WebkitBackgroundClip: 'text',
                                    WebkitTextFillColor: 'transparent',
                                    marginBottom: '16px',
                                    margin: 0,
                                }}>
                                    ERC20 Gas Payment Playground
                                </h1>
                                <p style={{ color: '#9ca3af', fontSize: '18px', margin: 0 }}>
                                    Experience ERC-20 gas payments with Smart Wallet SDK
                                </p>
                            </div>

                            <div style={{ marginBottom: '32px', paddingTop: '16px', display: 'flex', flexDirection: 'column', alignItems: 'center' }}>
                                <div style={{
                                    position: 'relative',
                                    width: '112px',
                                    height: '112px',
                                    margin: '0 auto 32px',
                                    display: 'flex',
                                    alignItems: 'center',
                                    justifyContent: 'center',
                                }}>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #FF3B57, #ec4899)',
                                        filter: 'blur(12px)',
                                        opacity: 0.7,
                                    }}></div>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #FF3B57, #ec4899)',
                                        display: 'flex',
                                        alignItems: 'center',
                                        justifyContent: 'center',
                                        color: 'white',
                                    }}>
                                        <div style={{ width: '48px', height: '48px', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                                            <WalletIcon />
                                        </div>
                                    </div>
                                </div>

                                <button
                                    onClick={handleCreateAccount}
                                    disabled={isLoading}
                                    style={{
                                        ...buttonStyles.primary,
                                        opacity: isLoading ? 0.7 : 1,
                                        transform: isLoading ? 'none' : 'scale(1)',
                                        margin: '0 auto',
                                    }}
                                    onMouseEnter={(e) =>
                                    {
                                        if (!isLoading) {
                                            e.target.style.transform = 'scale(1.05)'
                                        }
                                    }}
                                    onMouseLeave={(e) =>
                                    {
                                        if (!isLoading) {
                                            e.target.style.transform = 'scale(1)'
                                        }
                                    }}
                                >
                                    {isLoading ? (
                                        <>
                                            <LoaderIcon />
                                            Creating Account...
                                        </>
                                    ) : (
                                        <>
                                            Create Account
                                            <ChevronRightIcon />
                                        </>
                                    )}
                                </button>
                            </div>
                        </div>
                    )}

                    {state === "account-created" && (
                        <div style={{ textAlign: 'center', width: '100%' }}>
                            <div style={{ marginBottom: '24px' }}>
                                <div style={{
                                    position: 'relative',
                                    width: '80px',
                                    height: '80px',
                                    margin: '0 auto 16px',
                                }}>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #10b981, #059669)',
                                        filter: 'blur(12px)',
                                        opacity: 0.7,
                                    }}></div>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #10b981, #059669)',
                                        display: 'flex',
                                        alignItems: 'center',
                                        justifyContent: 'center',
                                        color: 'white',
                                    }}>
                                        <WalletIcon />
                                    </div>
                                </div>
                                <h2 style={{
                                    fontSize: '36px',
                                    fontWeight: 'bold',
                                    background: 'linear-gradient(to right, white, #d1d5db)',
                                    WebkitBackgroundClip: 'text',
                                    WebkitTextFillColor: 'transparent',
                                    marginBottom: '16px',
                                    margin: 0,
                                }}>
                                    Account Created Successfully!
                                </h2>
                                <p style={{ color: '#9ca3af', margin: 0 }}>
                                    Your smart wallet is ready for gasless transactions
                                </p>
                            </div>

                            <div style={{
                                background: 'rgba(0, 0, 0, 0.5)',
                                borderRadius: '12px',
                                padding: '24px',
                                border: '1px solid rgba(255, 59, 87, 0.2)',
                                boxShadow: 'inset 0 2px 4px rgba(255, 59, 87, 0.05)',
                                marginBottom: '48px',
                            }}>
                                <div style={{
                                    display: 'grid',
                                    gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))',
                                    gap: '24px',
                                    textAlign: 'left',
                                }}>
                                    <div>
                                        <label style={{
                                            fontSize: '14px',
                                            color: '#6b7280',
                                            marginBottom: '4px',
                                            display: 'block',
                                        }}>
                                            Wallet Address
                                        </label>
                                        <p style={{
                                            color: 'white',
                                            fontFamily: 'monospace',
                                            fontSize: '14px',
                                            wordBreak: 'break-all',
                                            background: 'rgba(0, 0, 0, 0.3)',
                                            padding: '10px',
                                            borderRadius: '6px',
                                            border: '1px solid #374151',
                                            margin: 0,
                                        }}>
                                            {address}
                                        </p>
                                    </div>
                                    <div>
                                        <label style={{
                                            fontSize: '14px',
                                            color: '#6b7280',
                                            marginBottom: '4px',
                                            display: 'block',
                                        }}>
                                            Network
                                        </label>
                                        <p style={{
                                            color: 'white',
                                            background: 'rgba(0, 0, 0, 0.3)',
                                            padding: '10px',
                                            borderRadius: '6px',
                                            border: '1px solid #374151',
                                            display: 'flex',
                                            alignItems: 'center',
                                            margin: 0,
                                        }}>
                                            <span style={{
                                                width: '8px',
                                                height: '8px',
                                                background: '#10b981',
                                                borderRadius: '50%',
                                                marginRight: '8px',
                                            }}></span>
                                            Base Sepolia
                                        </p>
                                    </div>
                                </div>
                            </div>

                            <button
                                onClick={handleSendTransaction}
                                disabled={isLoading || Number(balance) <= 0}
                                style={{
                                    ...buttonStyles.primary,
                                    padding: '20px 32px',
                                    opacity: isLoading ? 0.7 : (Number(balance) <= 0 ? 0.5 : 1),
                                    background: Number(balance) <= 0 ? '#23272e' : buttonStyles.primary.background,
                                    color: Number(balance) <= 0 ? '#9ca3af' : buttonStyles.primary.color,
                                    margin: '0 auto',
                                    cursor: isLoading ? 'not-allowed' : (Number(balance) <= 0 ? 'not-allowed' : 'pointer'),
                                }}
                                onMouseEnter={(e) =>
                                {
                                    if (!isLoading && Number(balance) > 0) {
                                        e.target.style.transform = 'scale(1.05)'
                                    }
                                }}
                                onMouseLeave={(e) =>
                                {
                                    if (!isLoading && Number(balance) > 0) {
                                        e.target.style.transform = 'scale(1)'
                                    }
                                }}
                            >
                                {isLoading ? (
                                    <>
                                        <LoaderIcon />
                                        Sending Transaction...
                                    </>
                                ) : (
                                    <>
                                        {Number(balance) > 0 ? (
                                            <>
                                                <SendIcon />
                                                Send Transaction
                                            </>
                                        ) : (
                                            <>
                                                <SendIcon />
                                                Insufficient Balance
                                            </>
                                        )}
                                    </>
                                )}
                            </button>
                        </div>
                    )}

                    {state === "transaction-sent" && (
                        <div style={{ textAlign: 'center' }}>
                            <div style={{ marginBottom: '24px' }}>
                                <div style={{
                                    position: 'relative',
                                    width: '80px',
                                    height: '80px',
                                    margin: '0 auto 16px',
                                }}>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #FF3B57, #ec4899)',
                                        filter: 'blur(12px)',
                                        opacity: 0.7,
                                    }}></div>
                                    <div style={{
                                        position: 'absolute',
                                        inset: 0,
                                        borderRadius: '50%',
                                        background: 'linear-gradient(to right, #FF3B57, #ec4899)',
                                        display: 'flex',
                                        alignItems: 'center',
                                        justifyContent: 'center',
                                        color: 'white',
                                    }}>
                                        <SendIcon />
                                    </div>
                                </div>
                                <h2 style={{
                                    fontSize: '36px',
                                    fontWeight: 'bold',
                                    background: 'linear-gradient(to right, white, #d1d5db)',
                                    WebkitBackgroundClip: 'text',
                                    WebkitTextFillColor: 'transparent',
                                    marginBottom: '16px',
                                    margin: 0,
                                }}>
                                    Transaction Sent!
                                </h2>
                                <p style={{ color: '#9ca3af', margin: 0 }}>
                                    Your ERC-20 gas payment transaction has been submitted to the network
                                </p>
                            </div>

                            <div style={{
                                background: 'rgba(0, 0, 0, 0.5)',
                                borderRadius: '12px',
                                padding: '24px',
                                border: '1px solid rgba(255, 59, 87, 0.2)',
                                boxShadow: 'inset 0 2px 4px rgba(255, 59, 87, 0.05)',
                                marginBottom: '48px',
                            }}>
                                <div style={{ display: 'flex', flexDirection: 'column', gap: '20px' }}>
                                    <div style={{ textAlign: 'left' }}>
                                        <label style={{
                                            fontSize: '14px',
                                            color: '#6b7280',
                                            marginBottom: '4px',
                                            display: 'block',
                                        }}>
                                            Transaction Hash
                                        </label>
                                        <div style={{
                                            display: 'flex',
                                            alignItems: 'center',
                                            background: 'rgba(0, 0, 0, 0.3)',
                                            padding: '10px',
                                            borderRadius: '6px',
                                            border: '1px solid #374151',
                                        }}>
                                            <p style={{
                                                color: 'white',
                                                fontFamily: 'monospace',
                                                fontSize: '14px',
                                                wordBreak: 'break-all',
                                                margin: 0,
                                                flex: 1,
                                            }}>
                                                {transactionHash}
                                            </p>
                                            <button style={{
                                                background: 'transparent',
                                                border: 'none',
                                                color: '#9ca3af',
                                                cursor: 'pointer',
                                                padding: '4px',
                                                marginLeft: '8px',
                                            }}
                                                onClick={() => window.open(`https://sepolia.basescan.org/tx/${transactionHash}`, "_blank")}
                                            >
                                                <ExternalLinkIcon />
                                            </button>
                                        </div>
                                    </div>
                                </div>
                            </div>

                            <div style={{
                                display: 'flex',
                                flexDirection: 'column',
                                gap: '16px',
                                justifyContent: 'center',
                                marginBottom: '24px',
                            }}>
                                <div style={{
                                    display: 'flex',
                                    flexDirection: 'row',
                                    gap: '16px',
                                    justifyContent: 'center',
                                    flexWrap: 'wrap',
                                }}>
                                    <button
                                        style={buttonStyles.secondary}
                                        onClick={() => window.open(`https://relay.gelato.digital/tasks/status/${userOpId}`, "_blank")}
                                        onMouseEnter={(e) =>
                                        {
                                            e.target.style.background = '#FF3B57'
                                            e.target.style.color = 'white'
                                        }}
                                        onMouseLeave={(e) =>
